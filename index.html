        // Three.js Crystal Animation

        const canvas = document.getElementById('prism-canvas');

        // Create multi-faceted crystal geometry

        // Main crystal (octahedron for gem-like appearance)

        const mainGeometry = new THREE.OctahedronGeometry(1.8, 0);

        const mainMaterial = new THREE.MeshPhongMaterial({

            color: 0x6366f1,

            emissive: 0x6366f1,

            emissiveIntensity: 0.8,

            transparent: true,

            opacity: 0.7,

            shininess: 100,

            specular: 0xffffff

        });

        

        const mainCrystal = new THREE.Mesh(mainGeometry, mainMaterial);

        crystalGroup.add(mainCrystal);


        // Inner glowing core

        const coreGeometry = new THREE.SphereGeometry(0.5, 32, 32);

        const coreMaterial = new THREE.MeshBasicMaterial({

            color: 0xa855f7,

            transparent: true,

            opacity: 0.9

        });

        const core = new THREE.Mesh(coreGeometry, coreMaterial);

        crystalGroup.add(core);


        // Light rays (cleaner, sharper)

        const rayGroup = new THREE.Group();

        for (let i = 0; i < 6; i++) {

            const rayGeometry = new THREE.CylinderGeometry(0.02, 0.001, 8, 8);

            const rayMaterial = new THREE.MeshBasicMaterial({

                color: 0xffffff,

                transparent: true,

                opacity: 0.6,

                blending: THREE.AdditiveBlending

            });

            const ray = new THREE.Mesh(rayGeometry, rayMaterial);

            

            // Position rays radiating outward

            const angle = (i * Math.PI * 2) / 6;

            ray.rotation.z = angle;

            ray.position.x = Math.cos(angle) * 0.5;

            ray.position.y = Math.sin(angle) * 0.5;

            

            rayGroup.add(ray);

        }

        crystalGroup.add(rayGroup);


        // Outer wireframe for definition

        const edges = new THREE.EdgesGeometry(mainGeometry);

        const lineMaterial = new THREE.LineBasicMaterial({ 

            color: 0xffffff,

            transparent: true,

            opacity: 0.6,

            linewidth: 2

        });

        const wireframe = new THREE.LineSegments(edges, lineMaterial);

        crystalGroup.add(wireframe);


        // Add secondary smaller crystals orbiting

        const smallCrystals = [];

        for (let i = 0; i < 3; i++) {

            const smallGeo = new THREE.TetrahedronGeometry(0.3, 0);

            const smallMat = new THREE.MeshPhongMaterial({

                color: 0xa855f7,

                emissive: 0xa855f7,

                emissiveIntensity: 0.4,

                transparent: true,

                opacity: 0.5

            });

            const smallCrystal = new THREE.Mesh(smallGeo, smallMat);

            smallCrystals.push(smallCrystal);

            crystalGroup.add(smallCrystal);

        }


        scene.add(crystalGroup);


        // Particle system

        const particleCount = 100;

        const particlesGeometry = new THREE.BufferGeometry();

        const particlePositions = [];

        const particleVelocities = [];

        

        for (let i = 0; i < particleCount; i++) {

            particlePositions.push(

                (Math.random() - 0.5) * 10,

                (Math.random() - 0.5) * 10,

                (Math.random() - 0.5) * 10

            );

            particleVelocities.push({

                x: (Math.random() - 0.5) * 0.02,

                y: (Math.random() - 0.5) * 0.02,

                z: (Math.random() - 0.5) * 0.02

            });

        }

        

        particlesGeometry.setAttribute('position', new THREE.Float32BufferAttribute(particlePositions, 3));

        

        const particleMaterial = new THREE.PointsMaterial({

                smallColor.setHSL(smallHue, 1.0, 0.5);

                crystal.material.color.copy(smallColor);

                crystal.material.emissive.copy(smallColor);

                crystal.material.emissiveIntensity = 0.6;

            });

            

            // Animate particles

            const positions = particles.geometry.attributes.position.array;

            for (let i = 0; i < particleCount; i++) {

                const i3 = i * 3;

                positions[i3] += particleVelocities[i].x;

                positions[i3 + 1] += particleVelocities[i].y;

                positions[i3 + 2] += particleVelocities[i].z;

                

                // Reset particles that drift too far

                if (Math.abs(positions[i3]) > 5) particleVelocities[i].x *= -1;

                if (Math.abs(positions[i3 + 1]) > 5) particleVelocities[i].y *= -1;

                if (Math.abs(positions[i3 + 2]) > 5) particleVelocities[i].z *= -1;

            }

            particles.geometry.attributes.position.needsUpdate = true;

            

            // Rotate particle cloud slowly

            particles.rotation.y = time * 0.1;

            

            // Animate lights

            light1.intensity = 3 + Math.sin(time * 2) * 1;

            light2.intensity = 3 + Math.sin(time * 2 + Math.PI) * 1;

            light3.intensity = 2 + Math.sin(time * 3) * 0.5;

            

            renderer.render(scene, camera);

        }

        animate();


        // Responsive resize

        document.querySelectorAll('a[href^="#"]').forEach(anchor => {

            anchor.addEventListener('click', function (e) {

                e.preventDefault();

                const target = document.querySelector(this.getAttribute('href'));

                if (target) {

                    target.scrollIntoView({ behavior: 'smooth' });

                }

            });

        });

    </script>

</body>
